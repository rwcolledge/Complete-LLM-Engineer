\chapter{Mathematics From Zero}
\label{chap:math-from-zero}
\chapterintro{This chapter begins the mathematics required for AI without assuming prior knowledge.}
\learningobjectives{\begin{itemize}\item Understand numbers and operations.\item Explain variables.\item Connect arithmetic to programming.\end{itemize}}
\section{Numbers}A number represents quantity. In AI, numbers represent data, model parameters, probabilities, losses, and predictions.
\section{Addition}\[2+3=5\]
Addition combines quantities.
\section{Variables}A variable is a symbol that stands for a value. For example, if \(x=3\), then \(x+2=5\).
\section{Python Example}\begin{lstlisting}
x = 3
print(x + 2)
\end{lstlisting}
\section{Exercises}\begin{exercise}If \(x=10\), compute \(x+7\).\end{exercise}\begin{solution}\(x+7=10+7=17\).\end{solution}
